函数

1.日期函数
        1.current_date();                                             当前日期
        2.current_time();                                             当前时间
        3.current_temetamp();                                   当前时间戳
        4.date(datetime);                                           返回 datetime 函数的日期部分
        5.date_add(date, interval d_value_d_type);   在 date 中加上日期或时间
        6.date_sub(date, interval d_value_d_type);   在 date 上减去一个时间
        7.datediff(date1, date2);                                两个日期差（天）date1 - date2
        8.now();                                                         当前时间

2.字符串函数
        1.charset(str);                                                返回字符串字符集
        2.concat(string2, [, ...]);                                  连接字符串
        3.instr(string, substring);                               返回 substring 在 string 中出现的位置，没有返回 0。
        4.ucase(string2);                                            转换为 大写
        5.lcase(string2);                                             转换为小写
        6.left(string2, length);                                    从 string2 中的左端起去 length 个字符
        7.length(string);                                             string 的长度
        8.replace(str, search_str, replace_str);            在 string 中用 replace_str 替换 search_str 
        9.strcmp(string1, string2);                             逐字符比较两字符串的大小
        10.substring(str, position [, length]);             从 str 的 position 开始，取 length 个字符
        11. ltrim(string)/ rtrim(string)/ trim(string);  去除前空格 、 后空格

3.数学函数
        1.abs(namber2);                                            绝对值
        2.bin(decimal_num);                                      十进制转二进制
        3.ceiling(name2)                                            向上取整
        4.conv(number, from_base, to_case);            进制转换
        5.floor(number2);                                          向下取整
        6.format(number, decimal_places);               保留小数位数
        7.hex(decimalNumber);                                 转换成 16 进制
        8.rand();                                                         返回随机浮点值， 范围 0-1  [0, 1);
        9.mod(numberator, denominator);               余数

4.其他常用函数
        1.user();                                                         查询当前用户
        2.md5(str);                                                    可以对一个字符串进行 md5 加密，加密后得到一个 32 位字符串
        3.database();                                                 显示当前正在使用的数据库
        4.password();                                                对 mysql 用户加密
        5.ifnull(val1, val2);                                         如果 val1 位 null, 返回 val2， 否则返回 val1 的值
