  1 #include <iostream>
  2 using namespace std;
  3
  4 int main()
  5 {

  9
10     //注意这里的这个成员函数必须使用，用来取消掉十进制的格式设置，否则下面的格式输出无效
11     cout.unsetf(ios::dec);   //终止十进制的格式设置
12
13     cout.setf(ios::hex);
14     cout << "hex:" << a << endl;
15
16     cout.unsetf(ios::hex);   //终止十六进制的格式设置
17

19     cout << "oct:" << a << endl;
20
21     cout.unsetf(ios::oct);   //终止八进制的格式设置
22
23     std::cout << "Hello World!" << endl;
24
25 ................................................................................................
26
27     char *p = "China";
28
29     cout.width(10);         //设置字宽为 10
30     cout << p << endl;
31
32     cout.width(10);            //成员函数 width(n) 和 控制符 setw(n) 只对其后面的第一个输出项有效，所以需要在输出每一项前都调用一次 width(n)
33     cout.fill('*');         //用字符 * 填充
34     cout << p << endl;

36 ................................................................................................
37
38     double pi = 22.0 / 7.0;
39
40     cout.setf(ios::scientific);   //指定为科学计数法输出
41     cout << "pi = ";
42     cout << pi << endl;
43
44     cout.unsetf(ios::scientific);   //终止科学计数法设置
45
46     cout.setf(ios::fixed);      //指定用定点形式输出
47     cout.width(12);
48     cout.setf(ios::showpos);    //正数输出 “+” 号
49     cout.setf(ios::internal);   //符号出现在最左侧, 如果没有的话会出现在数字的最左侧

51
52     cout << pi << endl;
